# C600系列(client端启动方式修改为vllm bench serve， bs input output 以及num-prompts对应关系与C500系列保持一致)

# 环境变量配置(目前测试只有单机)
单机模型环境变量:
ulimit -n 65536 && export MACA_SMALL_PAGESIZE_ENABLE=1 && export TRITON_ENABLE_MACA_OPT_MOVE_DOT_OPERANDS_OUT_LOOP=1 && export TRITON_ENABLE_MACA_CHAIN_DOT_OPT=1 && export MACA_DIRECT_DISPATCH=1 && export CUDA_VISIBLE_DEVICES=0,1,2,3

单机w8a8模型环境变量:
ulimit -n 65536 && export MACA_SMALL_PAGESIZE_ENABLE=1 && export TRITON_ENABLE_MACA_OPT_MOVE_DOT_OPERANDS_OUT_LOOP=1 && export TRITON_ENABLE_MACA_CHAIN_DOT_OPT=1 && export MACA_DIRECT_DISPATCH=1 && export TRITON_DISABLE_MACA_OPT_MMA_PREFETCH=1 && export TRITON_ENABLE_MACA_COMPILER_INT8_OPT=True && export TRITON_ENABLE_ELEMENTWISE_PK_FMA_OPT=True && export CUDA_VISIBLE_DEVICES=0,1,2,3,4,5,6,7

# 个别模型特殊参数设置
1. gpu_memory_vlaue 默认设置0.9，Qwen3-VL-235B-A22B-Instruct_W8A8 需设置为0.93，如遇显存不够可以下调至0.85
2. max_model_len设置不能超过模型本身长度，建议Qwen3与GLM模型设置32768，其它设置20480
3. mtp模型需要设置mtp参数--speculative_config '{"method":"mtp","num_speculative_tokens":2,"model":"${mtp_path}"}'

# 启动命令
## 多模态模型
### server端
vllm serve <模型路径> -tp <卡数> -pp <机器数目> --trust-remote-code --distributed-executor-backend mp --max-model-len 32768 --no-enable-prefix-caching --max-num-batched-tokens 8192 --dtype bfloat16 --gpu-memory-utilization 0.95 --port 25535 --mm-processor-cache-gb 0

### client端WW
vllm bench serve --model <模型路径> --dataset-name random-mm --random-mm-base-items-per-request 1 --random-mm-num-mm-items-range-ratio 0 --random-mm-limit-mm-per-prompt '{"image": 1, "video": 0}' --random-mm-bucket-config '[[1080, 1920, 1], 1.0]' --max-concurrency <concurrency> --random-input-len <input> --random-output-len <output> --num-prompts <prompts> --trust-remote-code --ignore-eos --backend openai-chat --endpoint /v1/chat/completions --port 25535

## 非多模态模型
### server端
vllm serve <模型路径> -tp <卡数> -pp <机器数目> --trust-remote-code --distributed-executor-backend mp --max-model-len 20480 --no-enable-prefix-caching --max-num-batched-tokens 8192 --dtype bfloat16 --gpu-memory-utilization 0.9 --port 25535

### client端
vllm bench serve --model <模型路径> --dataset-name random --random-input-len <input> --random-output-len <output> --num-prompts <prompts> --trust-remote-code --max-concurrency <concurrency> --ignore-eos --port 25535
